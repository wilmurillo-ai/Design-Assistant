# Memory Plus - 环境配置示例
# 复制这份文件为 memory-plus.env 并填入实际值

# ═══════════════════════════════════════════════════════
# Milvus 向量数据库配置
# ═══════════════════════════════════════════════════════

# Milvus 连接地址
# Docker Desktop/WSL2 环境：http://host.docker.internal:19530
# Linux 直接部署：http://localhost:19530
# 网络配置参考：https://milvus.io/docs/architecture_overview.md
MILVUS_URL=http://host.docker.internal:19530

# Milvus 集合名称（可自定义）
RAG_COLLECTION=openclaw_memory

# Milvus 认证（生产环境建议启用）
# MILVUS_USER=your_username
# MILVUS_PASSWORD=your_password


# ═══════════════════════════════════════════════════════
# Ollama 嵌入模型配置
# ═══════════════════════════════════════════════════════

# Ollama 服务地址
# Docker Desktop/WSL2 环境：http://host.docker.internal:11434
# Linux 直接部署：http://localhost:11434
OLLAMA_URL=http://host.docker.internal:11434

# 嵌入模型（需要提前用 ollama pull 下载）
# 推荐：bge-m3（支持中英日韩等 100+ 语言，1024维）
# 其他选择：nomic-embed-text（英文），mxbai-embed-large（多语言）
OLLAMA_MODEL=bge-m3
