def loading_spinner():

        # 显示旋转动画和百分比
        sys.stdout.write(f"\r执行中... {chars[i % 4]} {percent:.1f}%")
        sys.stdout.flush()
        if i < total:
            time.sleep(0.1)
    print("\r执行完成！                              ")
