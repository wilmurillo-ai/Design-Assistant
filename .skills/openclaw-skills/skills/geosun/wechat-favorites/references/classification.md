# 收藏分类算法

## 多标签分类

每篇文章可同时归属多个分类。

```python
CATEGORY_KEYWORDS = {
    'biomed': [
        '创新药', '临床试验', '靶点', '管线', '肿瘤', '免疫',
        'ADC', 'CAR-T', '单抗', '双抗', 'mRNA', '疫苗',
        '阿尔茨海默', '帕金森', '脑机接口', '神经',
        '药明', '恒瑞', '百济', '信达', '康方',
    ],
    'ai': [
        'GPT', 'OpenAI', '大模型', 'LLM', 'Agent',
        'DeepSeek', 'Claude', 'RAG', '多模态',
        '芯片', 'GPU', 'AI', '人工智能',
    ],
    'invest': [
        'IPO', '融资', '估值', '科创板', '港股',
        '并购', '基金', '投资', 'VC', 'PE',
        '高瓴', '红杉', 'IDG', '启明',
    ],
    'science': [
        '神经科学', '生物学', '物理学', '论文',
        'CNS', 'Nature', 'Science', 'Cell',
        '研究', '学术',
    ],
    'business': [
        '企业', '战略', '商业模式', '行业',
        '竞争', '市场', '增长', '转型',
    ],
    'lifestyle': [
        '健康', '养生', '心理', '职场', '教育',
        '文化', '旅行', '读书',
    ],
    'media': [
        '新闻', '媒体', '评论', '热点',
    ],
    'politics': [
        '中美', '国际', '地缘', '政策',
    ],
}

def classify_article(title, desc):
    text = f"{title} {desc}".lower()
    tags = []
    for cat, keywords in CATEGORY_KEYWORDS.items():
        for kw in keywords:
            if kw.lower() in text:
                tags.append(cat)
                break
    return tags or ['other']
```
