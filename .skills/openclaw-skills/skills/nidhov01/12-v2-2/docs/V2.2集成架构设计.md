### 1. 数据获取层 (Data Layer)

#### 1.1 DataFetcherManager (多数据源管理器)

**职责**: 管理多个数据源，实现自动故障切换

**设计参考**: daily_stock_analysis的DataFetcherManager

**数据源优先级**:
```
1. Efinance (priority=0)  - 主数据源，速度快
2. AKShare (priority=1)   - 备用数据源，免费稳定
3. Tushare (priority=2)   - 第三备用（需token）
```

**核心功能**:
```python
class DataFetcherManager:
    def __init__(self):
        self.fetchers = [
            EfinanceFetcher(priority=0),
            AkshareFetcher(priority=1),
            TushareFetcher(priority=2)
        ]
        self.circuit_breaker = CircuitBreaker()

    def get_daily_data(self, code, days=30):
        """自动故障切换获取日线数据"""
        for fetcher in self.fetchers:
            if not self.circuit_breaker.is_available(fetcher.name):
                continue

            try:
                df = fetcher.get_daily_data(code, days)
                self.circuit_breaker.record_success(fetcher.name)
                return df
            except Exception as e:
                self.circuit_breaker.record_failure(fetcher.name)
                logger.warning(f"{fetcher.name}失败，切换下一个")

        raise DataFetchError("所有数据源均失败")
```

**熔断机制**:
- 连续失败3次 → 熔断5分钟
- 半开状态试探性请求
- 成功后恢复正常

#### 1.3 ChipDistribution (筹码分布)

**职责**: 计算筹码集中度和获利盘比例

**数据结构**:
```python
@dataclass
class ChipDistribution:
    code: str
    date: str
    profit_ratio: float          # 获利盘比例(0-1)
    avg_cost: float              # 平均成本
    concentration_90: float      # 90%筹码集中度
    concentration_70: float      # 70%筹码集中度
```

**计算逻辑**:
```python
def calculate_chip_distribution(df, current_price):
    """
    基于价格分布计算筹码集中度

    算法：
    1. 将价格区间划分为100个档位
    2. 计算每个档位的成交量和持仓天数
    3. 找到90%筹码集中的价格区间
    4. 计算集中度 = (高-低) / 低
    5. 计算获利盘比例 = 价格 < 当前价 的筹码占比
    """
    # 价格分档
    price_bins = np.linspace(df['low'].min(), df['high'].max(), 100)

    # 成交量分布
    volume_dist = []
    for i in range(len(price_bins) - 1):
        mask = (df['close'] >= price_bins[i]) & (df['close'] < price_bins[i+1])
        vol = df.loc[mask, 'volume'].sum()
        volume_dist.append(vol)

    # 90%筹码集中度
    total_vol = sum(volume_dist)
    cum_vol = 0
    for i, vol in enumerate(sorted(volume_dist, reverse=True)):
        cum_vol += vol
        if cum_vol >= 0.9 * total_vol:
            concentration_90 = (i + 1) / 100
            break

    # 获利盘比例
    profit_ratio = (df['close'] < current_price).sum() / len(df)

    return ChipDistribution(
        concentration_90=concentration_90,
        profit_ratio=profit_ratio
    )
```

**判断逻辑**:
```python
def analyze_market_environment():
    """
    判断市场环境

    Returns:
        - 'strong': 强势市场
        - 'neutral': 震荡市场
        - 'weak': 弱势市场
    """
    # 1. 获取主要指数
    indices = get_main_indices(['000001', '399001', '399006'])  # 上证/深证/创业板

    # 2. 判断指数趋势
    ma_trend = all([
        idx['current'] > idx['ma5'] > idx['ma10'] > idx['ma20']
        for idx in indices
    ])

    # 3. 计算涨跌家数比
    up_down_ratio = market_stats['up_count'] / max(market_stats['down_count'], 1)

    # 4. 分类判断
    if ma_trend and up_down_ratio > 2:
        return 'strong'
    elif not ma_trend and up_down_ratio < 0.5:
        return 'weak'
    else:
        return 'neutral'
```

```python
def calculate_volume_ratio_signal(realtime_quote):
    """
    量比因子计算

    评分标准:
    - 量比 1.0-2.0: 正常，得分5分
    - 量比 2.0-3.0: 放量，得分7分
    - 量比 >3.0: 巨量，得分4分（风险）
    - 量比 <1.0: 缩量，得分3分
    """
    vr = realtime_quote.volume_ratio

    if 1.0 <= vr < 2.0:
        return 5.0
    elif 2.0 <= vr < 3.0:
        return 7.0
    elif vr >= 3.0:
        return 4.0  # 巨量风险
    else:
        return 3.0  # 缩量
```

**新增因子2: 筹码集中度 (5%)**

```python
def calculate_chip_concentration_signal(chip_dist):
    """
    筹码集中度因子计算

    评分标准:
    - 集中度 <8%: 高度集中，得分9分
    - 集中度 8-15%: 较集中，得分7分
    - 集中度 15-25%: 中等，得分5分
    - 集中度 >25%: 分散，得分3分
    """
    conc = chip_dist.concentration_90

    if conc < 0.08:
        return 9.0
    elif conc < 0.15:
        return 7.0
    elif conc < 0.25:
        return 5.0
    else:
        return 3.0
```

**完整信号计算**:

```python
def calculate_signal_v22(data, realtime_quote, chip_dist):
    """
    V2.2 8因子信号计算

    Returns:
        signal_score (0-10)
    """
    # 原有6因子
    momentum = calculate_momentum(data) * 0.25
    ma_dev = calculate_ma_deviation(data) * 0.20
    volume = calculate_volume(data) * 0.15
    volatility = calculate_volatility(data) * 0.10
    macd = calculate_macd(data) * 0.10
    bb = calculate_bollinger(data) * 0.10

    # 新增2因子
    vol_ratio = calculate_volume_ratio_signal(realtime_quote) * 0.05
    chip = calculate_chip_concentration_signal(chip_dist) * 0.05

    # 综合评分
    signal = (
        momentum + ma_dev + volume + volatility +
        macd + bb + vol_ratio + chip
    ).clip(0, 10)

    return signal
```

```python
def should_sell(position, data, current_price):
    """
    卖出决策（优先级机制）

    优先级:
    1. 止损 -5%
    2. 止盈 +10%
    3. 移动止盈
    4. MACD死叉
    5. 信号减弱 ≤6.5
    6. 持仓到期 60天
    """
    pnl = (current_price - position.buy_price) / position.buy_price

    # 1. 止损（最高优先级）
    if pnl < -0.05:
        return True, "止损"

    # 2. 止盈
    if pnl > 0.10:
        return True, "止盈"

    # 3. 移动止盈
    if pnl > 0.05 and current_price < position.max_price * 0.97:
        return True, "移动止盈"

    # 4. MACD死叉
    if is_macd_death_cross(data):
        return True, "MACD死叉"

    # 5. 信号减弱
    if position.signal <= 6.5:
        return True, "信号减弱"

    # 6. 持仓到期
    if position.holding_days > 60:
        return True, "持仓到期"

    return False, None
```

#### 6.1 EmailNotifier (邮件通知)

**职责**: 推送交易信号和风险警报

**功能模块**:
```python
class EmailNotifier:
    def __init__(self, config):
        self.smtp_server = config['smtp_server']
        self.smtp_port = config['smtp_port']
        self.sender = config['sender']
        self.password = config['password']
        self.receivers = config['receivers']

    def send_buy_signal(self, code, name, price, signal_score):
        """发送买入信号"""
        subject = f"🟢 买入信号: {name}({code})"
        body = f"""
买入信号触发！

股票: {name}({code})
买入价: {price:.2f}
信号评分: {signal_score:.2f}/10

建议:
- 止损价: {price * 0.95:.2f} (-5%)
- 目标价: {price * 1.10:.2f} (+10%)

时间: {datetime.now().strftime('%Y-%m-%d %H:%M')}
        """
        self._send(subject, body)

    def send_sell_signal(self, code, name, price, reason, pnl):
        """发送卖出信号"""
        subject = f"🔴 卖出信号: {name}({code})"
        body = f"""
卖出信号触发！

股票: {name}({code})
卖出价: {price:.2f}
卖出原因: {reason}
收益率: {pnl*100:.2f}%

时间: {datetime.now().strftime('%Y-%m-%d %H:%M')}
        """
        self._send(subject, body)

    def send_risk_alert(self, code, name, drawdown):
        """发送风险警报"""
        subject = f"⚠️ 风险警报: {name}({code})"
        body = f"""
风险警报触发！

股票: {name}({code})
当前回撤: {drawdown*100:.2f}

请及时关注！

时间: {datetime.now().strftime('%Y-%m-%d %H:%M')}
        """
        self._send(subject, body)

    def send_daily_report(self, positions, total_pnl):
        """发送每日汇总报告"""
        subject = f"📊 每日交易报告 {datetime.now().strftime('%Y-%m-%d')}"
        body = f"""
每日交易汇总

持仓数量: {len(positions)}
总收益率: {total_pnl*100:.2f}%

持仓明细:
{self._format_positions(positions)}

时间: {datetime.now().strftime('%Y-%m-%d %H:%M')}
        """
        self._send(subject, body)

    def _send(self, subject, body):
        """发送邮件"""
        import smtplib
        from email.mime.text import MIMEText

        msg = MIMEText(body, 'plain', 'utf-8')
        msg['Subject'] = subject
        msg['From'] = self.sender
        msg['To'] = ', '.join(self.receivers)

        with smtplib.SMTP(self.smtp_server, self.smtp_port) as server:
            server.starttls()
            server.login(self.sender, self.password)
            server.send_message(msg)
```

```python
    3. 初始化通知服务（可选）
    4. 遍历交易日
    5. 每日流程:
       - 获取市场环境
       - 获取实时行情
       - 计算筹码分布
       - 计算8因子信号
       - 执行买卖决策
       - 发送通知（可选）
    6. 计算性能指标
    7. 生成报告
    """
    # 初始化
    data_manager = DataFetcherManager()
    market_analyzer = MarketAnalyzer()
    notifier = EmailNotifier(config) if config['notification']['email']['enabled'] else None

    # 回测循环
    for date in trading_dates:
        # 1. 判断市场环境
        market_env = market_analyzer.analyze_environment(date)

        # 2. 遍历持仓
        for position in positions:
            # 获取实时行情
            realtime_quote = data_manager.get_realtime_quote(position.code)

            # 获取筹码分布
            chip_dist = data_manager.get_chip_distribution(position.code, date)

            # 计算8因子信号
            signal = calculate_signal_v22(
                position.data,
                realtime_quote,
                chip_dist
            )

            # 决策
            if should_sell(position, signal):
                execute_sell(position)
                if notifier:
                    notifier.send_sell_signal(...)

        # 3. 遍历观察列表
        for code in watch_list:
            # 获取数据
            data = data_manager.get_daily_data(code)
            realtime_quote = data_manager.get_realtime_quote(code)
            chip_dist = data_manager.get_chip_distribution(code, date)

            # 计算信号
            signal = calculate_signal_v22(data, realtime_quote, chip_dist)

            # 买入决策（动态阈值）
            if should_buy(signal, market_env):
                execute_buy(code)
                if notifier:
                    notifier.send_buy_signal(...)

    # 计算指标
    metrics = calculate_performance_metrics()

    return metrics
```
