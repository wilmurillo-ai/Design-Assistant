```python
# 在文件开头添加新函数
def calculate_volume_ratio_signal(df):
    """量比因子 - 5%权重"""
    if len(df) < 5:
        return 5.0

    # 计算量比
    recent_vol = df['volume'].iloc[-1]
    avg_vol_5d = df['volume'].tail(5).mean()

    if avg_vol_5d == 0:
        return 5.0

    volume_ratio = recent_vol / avg_vol_5d

    # 评分
    if 1.0 <= volume_ratio < 2.0:
        return 7.0
    elif 2.0 <= volume_ratio < 3.0:
        return 9.0
    elif volume_ratio >= 3.0:
        return 5.0  # 巨量，风险增加
    else:
        return 3.0  # 缩量


def calculate_chip_signal(df):
    """筹码集中度因子 - 5%权重（简化版）"""
    if len(df) < 20:
        return 5.0

    # 使用最近60天的价格波动作为筹码集中度代理指标
    prices = df['close'].tail(60)
    std_dev = prices.std()
    mean_price = prices.mean()

    if mean_price == 0:
        return 5.0

    # 变异系数 - 越小越集中
    cv = (std_dev / mean_price) * 100

    # 评分
    if cv < 5:
        return 9.0   # 高度集中
    elif cv < 10:
        return 7.0   # 较集中
    elif cv < 20:
        return 5.0   # 中等
    else:
        return 3.0   # 分散
```

### Step 2: 修改信号计算公式

找到signal计算部分，修改为：

```python
# 原V2.1: 6因子
# V2.2: 8因子

# 动量 (25%)
momentum_signal = (data['momentum'].clip(-0.15, 0.15) / 0.15 * 5 + 5) * 0.25

# 均线偏离 (20%)
ma_dev_signal = (1 - data['ma_dev'].clip(-0.1, 0.1) / 0.1 * 5) * 0.20

# 成交量 (15%)
vol_signal = ((data['vol_ratio'] - 1).clip(-0.5, 0.5) / 0.5 * 5 + 5) * 0.15

# 波动率 (10%)
volatility_signal = (1 - data['volatility'].clip(0, 0.1) / 0.1 * 5) * 0.10

# MACD (10%)
macd_signal = (data['macd_hist'].clip(-0.05, 0.05) / 0.05 * 5 + 5) * 0.10

# 布林带 (10%)
bb_signal = (data['bb_position'] * 10) * 0.10

# ===== 新增因子 =====
# 量比 (5%)
vr_signal = (calculate_volume_ratio_signal(data)) * 0.05

# 筹码集中度 (5%)
chip_signal = (calculate_chip_signal(data)) * 0.05

# 综合评分
data['signal'] = (
    momentum_signal + ma_dev_signal + vol_signal +
    volatility_signal + macd_signal + bb_signal +
    vr_signal + chip_signal  # 新增
).clip(0, 10)
```
