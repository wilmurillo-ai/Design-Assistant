> 本模板用于**一条主线下多个关联题材**的联合深度分析，与模板2（单题材）和模板3（两两对比）不同。

### 前置操作

```
1. 从 fullList 按主线归类相关题材（见主线识别规则）
2. 批量调用 subjectDetail4Free 获取每个题材详情
3. 跨题材提取全部个股，去重统计
4. 使用 api-reference.md 中的 fetch_quotes() 批量获取行情
5. Web 搜索该主线相关资讯
```

## 通用数据处理工具函数

以下函数供各模板的数据处理环节调用。行情获取使用 `api-reference.md` 中的 `fetch_quotes()` 统一脚本。

```python
import requests

BASE_URL = "https://group-api.itougu.com"
HEADERS = {"Content-Type": "application/json"}

# ===== API 调用 =====

def get_full_list():
    """获取题材库全量列表，按 orderNumber 排序"""
    resp = requests.post(f"{BASE_URL}/teach-hotspot/subject/fullList",
                         json={}, headers=HEADERS)
    result = resp.json()
    if result["code"] != 20000:
        raise Exception(f"fullList 错误: {result.get('msg', '未知')}")
    return sorted(result["data"]["rows"], key=lambda x: x.get("orderNumber", 999))

def get_subject_detail(subject_id):
    """获取单个题材详情"""
    resp = requests.post(f"{BASE_URL}/teach-hotspot/subject/subjectDetail4Free",
                         json={"subjectId": subject_id}, headers=HEADERS)
    result = resp.json()
    if result["code"] == 50000:
        return None  # 数据不存在
    if result["code"] != 20000:
        raise Exception(f"detail 错误: {result.get('msg', '未知')}")
    return result["data"]

# ===== 树状结构解析 =====

def parse_tree(node, level=0, stats=None):
    """递归解析 tableJson，统计分类/个股/高亮/掩码数"""
    if stats is None:
        stats = {'categories': 0, 'stocks': 0, 'highlights': 0,
                 'masked': 0, 'max_depth': 0}
    if node.get("nodeType") == 1 and level > 0:
        stats['categories'] += 1
    elif node.get("nodeType") == 2:
        stats['stocks'] += 1
        if node.get("highlight") == 1:
            stats['highlights'] += 1
        if '***' in (node.get("name") or ''):
            stats['masked'] += 1
    stats['max_depth'] = max(stats['max_depth'], level)
    for child in (node.get("children") or []):
        parse_tree(child, level + 1, stats)
    return stats

def extract_stocks(node, theme_name, result_dict):
    """递归提取个股到 result_dict，支持跨题材去重"""
    if node.get("nodeType") == 2 and node.get("stockCode"):
        sc = node["stockCode"]
        if sc not in result_dict:
            result_dict[sc] = {'name': node['name'], 'themes': [], 'highlight': 0}
        result_dict[sc]['themes'].append(theme_name)
        if node.get('highlight') == 1:
            result_dict[sc]['highlight'] = 1
    for child in (node.get("children") or []):
        extract_stocks(child, theme_name, result_dict)

def find_stock_in_tree(node, target):
    """在树中搜索目标个股（按名称或代码），返回 (found, is_highlight)"""
    if node.get("nodeType") == 2:
        name = node.get("name", "")
        code = (node.get("stockCode") or "").split(".")[0]
        if target in name or target == code:
            return True, node.get("highlight") == 1
    for child in (node.get("children") or []):
        found, highlighted = find_stock_in_tree(child, target)
        if found:
            return True, highlighted
    return False, False

# ===== 热度评分 =====

def heat_score(subject_item, detail=None):
    """计算题材热度评分 (0-100)"""
    score = 0
    order = subject_item.get("orderNumber", 999)
    if order <= 10: score += 40
    elif order <= 30: score += 25
    elif order <= 50: score += 15
    else: score += 5
    score += min(subject_item.get("upLimitNum", 0) * 10, 30)
    tag = subject_item.get("squareTag", 0)
    if tag == 2: score += 15
    elif tag == 1: score += 10
    if subject_item.get("rectangleTag"): score += 10
    if detail:
        if detail.get("dragonHeadFlag") == 1: score += 2.5
        if detail.get("middleTroopsFlag") == 1: score += 2.5
    return min(score, 100)
```
