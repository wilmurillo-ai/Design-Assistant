### 双模式解析

```python
def parse_case_info(self, user_input):
    # 1. 检测输入类型
    input_type = self._detect_input_type(user_input)
    # structured | mixed | simple
    
    # 2. 分级解析
    if input_type == 'structured':
        case_info = self._parse_structured_input(user_input)
    elif input_type == 'mixed':
        case_info = self._parse_mixed_input(user_input)
    else:
        case_info = self._parse_simple_input(user_input)
    
    # 3. 评估完整度
    completeness, missing = self._assess_completeness(case_info)
    
    # 4. 生成追问
    questions = self.generate_follow_up_questions(case_info)
    
    return case_info
```

### 智能追问

```python
def generate_follow_up_questions(self, case_info):
    questions = []
    missing = case_info.get('missing_fields', [])
    
    # 必填字段追问
    if '纠纷类型' in missing:
        questions.append('请问这是什么类型的纠纷？')
    
    # 根据纠纷类型追问特定信息
    dispute = case_info.get('纠纷类型', '')
    if '劳动' in dispute:
        if not case_info.get('工作年限'):
            questions.append('您工作了多长时间？')
        if not case_info.get('月工资'):
            questions.append('您的月工资是多少？')
    
    return questions
```
