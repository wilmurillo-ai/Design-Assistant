## 3. 多 Agent 视角 (P1)

### 设计

```python
# 修改 memory.py

class MemoryStore:
    def store(self, content: str, metadata: dict = None, 
              agent_perspective: str = None):
        """
        存储记忆
        
        Args:
            agent_perspective: 
                - None: 共享记忆 (默认)
                - "xiao-zhi": 小智视角
                - "xiao-liu": 小刘视角
        """
        if agent_perspective:
            metadata = metadata or {}
            metadata["agent_perspective"] = agent_perspective
        
        # 存储到向量库
        self._store_to_vector(content, metadata)
        
        # 如果是共享记忆，同步到图谱
        if not agent_perspective:
            self._sync_to_graph(content, metadata)
    
    def search(self, query: str, agent_perspective: str = None, **kwargs):
        """搜索记忆，支持按视角过滤"""
        filters = kwargs.get("filters", {})
        if agent_perspective:
            filters["agent_perspective"] = agent_perspective
        return self._search_vector(query, filters=filters, **kwargs)
```

### CLI 接口

```bash
# 存储到特定视角
python3 scripts/memory.py store "内容" --agent xiao-zhi

# 搜索特定视角
python3 scripts/memory.py search "查询" --agent xiao-zhi

# 查看共享记忆
python3 scripts/memory.py search "查询" --shared-only
```

## 5. 记忆质量自检 (P2)

### 设计

```python
# 增强 memory_health.py

AUTO_FIX_RULES = {
    "contradiction": {
        "action": "merge_or_flag",
        "threshold": 0.7,
        "auto_merge": False  # 需要确认
    },
    "outdated": {
        "action": "archive_or_update",
        "threshold_days": 90,
        "auto_archive": True
    },
    "redundant": {
        "action": "compress",
        "similarity_threshold": 0.85,
        "auto_compress": True
    },
    "orphaned": {
        "action": "link_or_delete",
        "auto_delete": False
    }
}

class MemoryHealthChecker:
    def auto_fix(self, dry_run: bool = True):
        """自动修复质量问题"""
        fixes = []
        
        # 1. 检测矛盾
        contradictions = self._detect_contradictions()
        fixes.extend(self._resolve_contradictions(contradictions, dry_run))
        
        # 2. 检测过时
        outdated = self._detect_outdated()
        fixes.extend(self._archive_outdated(outdated, dry_run))
        
        # 3. 检测冗余
        redundant = self._detect_redundant()
        fixes.extend(self._compress_redundant(redundant, dry_run))
        
        # 4. 检测孤立
        orphaned = self._detect_orphaned()
        fixes.extend(self._handle_orphaned(orphaned, dry_run))
        
        return fixes
```

### CLI 接口

# 自动修复 (执行)
python3 scripts/memory_health.py auto-fix --apply

# 生成报告
