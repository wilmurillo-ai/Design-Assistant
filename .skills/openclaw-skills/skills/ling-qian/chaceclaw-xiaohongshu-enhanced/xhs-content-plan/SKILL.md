### 选题优先级评估

```python
def score_topic(keyword, search_volume, competition, trend):
    """
    选题评分：综合搜索量、竞争度、趋势
    分数越高越好
    """
    # 搜索量权重 40%，竞争度权重 40%，趋势权重 20%
    score = (
        min(search_volume / 10000, 1.0) * 0.4 +
        (1 - min(competition / 1000, 1.0)) * 0.4 +
        trend * 0.2
    )
    return round(score * 100, 1)

# 评分标准
# ≥80: 黄金选题（必做）
# 60-79: 优质选题（优先做）
# 40-59: 普通选题（有空再做）
# <40: 避开（竞争太激烈或需求太小）
```

## 内容复盘框架

```python
def monthly_review(month_stats):
    """月度复盘"""
    
    print("=" * 50)
    print(f"📊 {month_stats['month']} 内容复盘")
    print("=" * 50)
    
    # 数据总览
    print(f"\n📈 基础数据：")
    print(f"  发布笔记：{month_stats['posts']} 篇")
    print(f"  总点赞：{month_stats['total_likes']:,}")
    print(f"  总收藏：{month_stats['total_collects']:,}")
    print(f"  总评论：{month_stats['total_comments']:,}")
    print(f"  净增粉丝：{month_stats['new_followers']:,}")
    
    # 内容效果分析
    print(f"\n📋 内容效果：")
    for i, post in enumerate(month_stats['top_posts'], 1):
        print(f"  {i}. {post['title'][:30]}... | 赞{post['likes']} 藏{post['collects']}")
    
    # 改进建议
    print(f"\n💡 下月改进方向：")
    if month_stats['avg_likes_per_post'] < 50:
        print("  - 标题需要优化，增加吸引力")
    if month_stats['engagement_rate'] < 3:
        print("  - 增加互动引导，提升评论率")
    if month_stats['viral_count'] == 0:
        print("  - 尝试蹭热点或策划更爆款的内容方向")
    
    return month_stats.get('next_month_focus', [])
```
