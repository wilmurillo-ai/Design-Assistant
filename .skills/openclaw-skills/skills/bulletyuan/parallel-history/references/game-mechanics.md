**地点对决策的影响**：

```python
def 计算地点影响(决策, 地点环境):
    影响 = {
        "成功率修正": 1.0,
        "风险修正": 1.0,
        "特殊选项": [],
        "特殊限制": []
    }

    # 气候影响
    if 决策.类型 == "军事行动" and 地点环境.气候.季节 == "冬季":
        影响.成功率修正 *= 0.7  # 冬季作战困难
        影响.风险修正 *= 1.3    # 风险增加

    # 地形影响
    if 决策.类型 == "防御战":
        if 地点环境.地理.地形 == "山地":
            影响.成功率修正 *= 1.3  # 山地易守
        elif 地点环境.地理.地形 == "平原":
            影响.成功率修正 *= 0.8  # 平原难守

    # 民风影响
    if 决策.类型 == "社会改革":
        if 地点环境.民风.传统 == "保守":
            影响.成功率修正 *= 0.7
            影响.特殊限制.append("保守势力阻力大")
        elif 地点环境.民风.传统 == "开放":
            影响.成功率修正 *= 1.2

    # 资源影响
    if 决策.需要资源 > 阵营.资源:
        if 地点环境.资源.农业 == "丰富":
            影响.成功率修正 *= 1.1  # 本地资源补充
        elif 地点环境.资源.农业 == "贫瘠":
            影响.成功率修正 *= 0.9  # 资源短缺加剧

    # 信息影响
    if 决策.类型 == "秘密行动":
        if 地点环境.信息.交通枢纽:
            影响.风险修正 *= 1.4  # 枢纽城市监控严
            影响.特殊限制.append("容易被发现")
        elif 地点环境.信息.边疆:
            影响.风险修正 *= 0.8  # 边疆监控弱
            影响.特殊选项.append("可逃往国外")

    return 影响
```

```python
  "气候趋势": "小冰河期/温暖期/稳定期",

  "疫病风险": 人口密度 × 卫生条件 × 国际交流度,
  "农业产出": 基础产出 × (1 - 灾害损失) × 技术进步加成
}

# 社会环境变化
社会环境 = {
  "阶级流动": 城市化率 × 教育普及 × 制度开放度,
  "人口结构": 出生率 - 死亡率 + 迁移率,
  "社会矛盾": (贫富差距 × 阶级固化) - (改革缓解 × 社会保障),
  "动员潜力": 人口基数 × 组织程度 × 意识形态认同
}

# 生产力环境变化
生产力环境 = {
  "技术水平": 基础水平 + (研发投入 × 时间) + (技术扩散 × 交流度),
  "生产方式": 小农经济 → 手工业 → 工场 → 工厂 (渐进演变),
  "经济形态": 自然经济 → 商品经济 → 市场经济 (渐进演变),
  "基础设施": 道路/铁路/通讯网络的扩张速度

# 文化环境变化
文化环境 = {
  "思想潮流": 传统思想 ← → 新兴思想 (竞争与融合),
```

### 🎲 概率系统（增强版）

**决策成功率计算**：

```python
def 计算成功率(决策, 环境, 阵营, 人物羁绊, 地点环境):
    # 基础概率
    if 决策.符合历史趋势:
        基础概率 = random(60, 80)
    elif 决策.逆历史潮流:
        基础概率 = random(20, 40)
    else:
        基础概率 = random(40, 60)

    # 四维环境修正
    环境修正 = 1.0
    if not 决策.物质基础可行(环境.生产力):
        环境修正 *= 0.3  # 物质条件不足，大幅降低
    if not 决策.社会基础可行(环境.社会):
        环境修正 *= 0.5
    if not 决策.文化基础可行(环境.文化):
        环境修正 *= 0.7

    # 🗺️ 地点环境修正（新增）
    地点影响 = 计算地点影响(决策, 地点环境)
    环境修正 *= 地点影响.成功率修正

    # 阵营实力修正
    阵营修正 = 阵营.影响力 / 100 × 阵营.资源 / 100

    # 人物羁绊修正
    羁绊修正 = 1.0
    for 盟友 in 决策.涉及人物:
        if 盟友.好感度 >= 80:
            羁绊修正 *= 1.2  # 强力盟友加成
        elif 盟友.好感度 <= -50:
            羁绊修正 *= 0.7  # 内部阻力

    # 历史惯性（变革阻力）
    历史惯性 = 决策.激进程度 × 环境.既得利益强度 × 0.01

    # 最终成功率
    成功率 = 基础概率 × 环境修正 × 阵营修正 × 羁绊修正 × (1 - 历史惯性)

    # 🗺️ 地点风险修正（新增）
    风险调整 = 地点影响.风险修正
    if 风险调整 > 1.0:
        # 高风险地点，成功率波动更大
        成功率 = 成功率 * random(0.7, 1.0)
    elif 风险调整 < 1.0:
        # 低风险地点，成功率更稳定
        成功率 = 成功率 * random(0.9, 1.1)

    return max(5, min(95, 成功率))  # 限制在5%-95%之间
```

### 📈 组织演变

**每回合变化**：
```
影响力 = 影响力 + (决策影响 × 随机因子)
稳定性 = 稳定性 - (外部压力 × 0.1) + (内部改革 × 0.05)
成员数 = 成员数 × (1 + 传播率 - 衰退率)
```

**组织状态判定**：
- `active`: 影响力 > 30，稳定性 > 40
- `underground`: 影响力 < 30，或被镇压
- `extinct`: 成员数 < 100，或影响力 = 0

### 📿 教派演变

**每回合变化**：
```
信徒数 = 信徒数 × (1 + 传播率)
教义纯度 = 教义纯度 - (制度化 × 0.05)
制度化 = 制度化 + (时间 × 0.02)
```

**教派状态判定**：
- `growing`: 传播率 > 3%
- `stable`: 传播率 1-3%
- `declining`: 传播率 < 1%
- `extinct`: 信徒数 < 1000

## 平衡机制

### 🎯 历史惯性（客观约束）

**定义**：历史趋势的抵抗力量，变革的物质与社会阻力

**三维约束模型**：

```python

    "物质阻力": {
        "技术限制": "生产力水平不足以支撑变革",
        "资源不足": "财政、人力、物资短缺",
        "基础设施": "交通、通讯、工业基础薄弱"
    },
    "社会阻力": {
        "既得利益": "旧制度的受益者抵抗",
        "阶级固化": "社会流动渠道被垄断",

    },

        "价值冲突": "新旧价值观剧烈对抗"
    }
}
```

**惯性计算公式**：

```python
def 计算历史惯性(决策, 环境):
    # 物质阻力
    物质阻力 = 0
    if 决策.所需技术水平 > 环境.生产力.技术水平:
        物质阻力 += (决策.所需技术水平 - 环境.生产力.技术水平) × 0.5
    if 决策.所需资源 > 阵营.资源:
        物质阻力 += (决策.所需资源 - 阵营.资源) / 阵营.资源 × 30

    # 社会阻力
    社会阻力 = 环境.社会.既得利益强度 × 决策.威胁既得利益程度 × 0.01
    社会阻力 += (100 - 环境.社会.动员潜力) × 决策.所需动员度 × 0.005

    # 文化阻力
    文化阻力 = (100 - 环境.文化.教育水平) × 决策.思想激进程度 × 0.003
    文化阻力 += 环境.文化.传统观念强度 × 决策.挑战传统程度 × 0.004

    return min(90, 物质阻力 + 社会阻力 + 文化阻力)
```

### ⚖️ 因果平衡

**原则**：
- 每个决策都有**即时代价**和**长期后果**
- 成功伴随风险，失败也有转机
- **不能脱离历史条件**，激进变革往往失败

**决策影响模型**：

