# -----------------------------------------------------------------------------
# 必需配置 (Required Configuration)
# -----------------------------------------------------------------------------

# 企业微信 CorpID
# 作用：标识企业身份的唯一ID
# 获取方式：企业微信管理后台 -> "我的企业" -> "企业信息" -> "企业ID"
WECOM_CORP_ID=your_corp_id_here

# 企业微信应用密钥
# 作用：调用企业微信API的密钥凭证
# 获取方式：企业微信管理后台 -> "应用管理" -> 选择应用 -> "Secret"
# 注意：请妥善保管此密钥，不要泄露或提交到代码仓库
WECOM_CORP_SECRET=your_corp_secret_here

# 企业微信应用 AgentID
# 作用：指定使用哪个应用发送消息
# 获取方式：企业微信管理后台 -> "应用管理" -> 选择应用 -> 顶部显示的"AgentId"
WECOM_AGENT_ID=1000004

# -----------------------------------------------------------------------------
# 可选配置 (Optional Configuration)
# -----------------------------------------------------------------------------

# 默认接收人 (touser)
# 作用：设置默认消息接收人
# 可选值：
#   - "@all" : 发送给所有用户
#   - "user1|user2|user3" : 发送给指定用户（多个用户用 | 分隔）
#   - "部门ID" : 发送给指定部门的所有成员（需配合 toparty 使用）
# 获取用户账号：企业微信管理后台 -> "通讯录" -> 可见范围内用户的"账号"
WECOM_TOUSER=@all

# 默认接收部门 (toparty)
# 作用：设置默认接收消息的部门
# 获取方式：企业微信管理后台 -> "通讯录" -> 部门名称旁边的 ID
# 注意：touser、toparty、totag 至少要有一个非空
WECOM_TOPARTY=

# 默认接收标签 (totag)
# 作用：设置默认接收消息的标签
# 获取方式：企业微信管理后台 -> "通讯录" -> 标签 -> 标签 ID
WECOM_TOTAG=

# -----------------------------------------------------------------------------
# 发送来源标识 (Sender Configuration)
# -----------------------------------------------------------------------------
# 作用：标识消息的发送来源，让接收者知道消息来自哪个服务/系统
# 示例值：
#   - "📦 数据备份服务"  : 定时备份任务
#   - "🚨 系统监控中心"  : 监控告警通知
#   - "🔄 CI/CD 流水线"  : 持续集成/部署
#   - "📊 数据分析平台"  : 数据报表
#   - "⏰ 定时清理任务"  : 定时任务
# 注意：此配置为默认发送来源，也可通过命令行 -s 参数覆盖
WECOM_SENDER=🎸 XiaoqiangClub
