## 🎯 使用示例

### 示例1：使用标准评估系统

```javascript
const StandardAssessmentSystem = require('./skills/standard-assessment/evaluate.js');

// 创建评估系统实例
const system = new StandardAssessmentSystem();

// 准备问卷答案
const answers = {
    // 基础信息
    q1: '<10人',                    // 组织规模
    q2: ['民事诉讼', '企业法务', '风险管理'],  // 业务领域
    q3: '基础办公软件',              // IT基础设施
    q4: '未开始',                   // 数字化程度
    q5: '了解基本概念',              // AI技术熟悉度
    
    // 深度评估
    d1: '<100条',                   // 知识库规模
    d2: '无标准化',                 // 标准化程度
    d3: '从不',                     // 跨部门协作
    d4: ['案例分析统计', '风险预警'],   // 数据分析需求
    d5: '偶尔需要',                 // 移动办公
    
    // 挑战与需求
    c1: '成本控制',                 // 最大挑战
    c2: ['降低成本', '提升效率', '风险控制'],  // 期望解决问题
    c3: '10-50万',                  // 预算规模
    c4: '3-6个月',                  // 实施周期
    c5: '支持'                      // 管理层支持
};

// 用户信息
const userInfo = {
    name: 'XX律师事务所',
    organization: 'XX律师事务所',
    industry: '法律服务',
    size: '小型律所（<10人）'
};

// 生成评估报告
system.generateReport(answers, userInfo).then(report => {
    // 输出JSON格式
    console.log('=== 评估报告（JSON） ===');
    console.log(JSON.stringify(report, null, 2));
    
    // 输出Markdown格式
    console.log('\n=== 评估报告（Markdown） ===');
    console.log(system.exportToMarkdown(report));
    
    // 保存为文件
    fs.writeFileSync(
        `评估报告-${userInfo.name}-${new Date().toISOString().split('T')[0]}.json`,
        JSON.stringify(report, null, 2)
    );
    
    fs.writeFileSync(
        `评估报告-${userInfo.name}-${new Date().toISOString().split('T')[0]}.md`,
        system.exportToMarkdown(report)
    );
});
```

## 🔧 快速开始

### 1. 安装依赖

```bash
cd /workspace/projects/agents/legal-ai-team/legal-ceo/workspace

npm init -y
npm install openai axios fs-extra
```

### 2. 运行示例

```bash
node skills/standard-assessment/evaluate.js
node skills/standard-implementation/navigation.js
```

### 3. 集成到现有系统

```javascript
const StandardAssessmentSystem = require('./skills/standard-assessment/evaluate.js');
const StandardImplementationNavigator = require('./skills/standard-implementation/navigation.js');

// 创建实例
const assessmentSystem = new StandardAssessmentSystem();
const implementationNavigator = new StandardImplementationNavigator();

// 使用系统
const answers = { /* 用户答案 */ };
const userInfo = { /* 用户信息 */ };

// 生成评估报告
const report = await assessmentSystem.generateReport(answers, userInfo);

// 生成实施计划
const plan = implementationNavigator.generateImplementationPlan('medium');

// 输出结果
console.log('综合评分：', report.assessment.overall.score);
console.log('适配等级：', report.assessment.overall.adaptationLevel.level);
console.log('预估成本：', plan.estimated.cost);
console.log('预估时间：', plan.estimated.time);
```
