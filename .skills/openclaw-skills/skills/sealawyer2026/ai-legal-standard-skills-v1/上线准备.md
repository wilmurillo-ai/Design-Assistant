## 🎯 使用示例

```javascript
const StandardSkillsPackage = require('./skills/index.js');

// 创建技能包实例
const skillsPackage = new StandardSkillsPackage();

// 1. 评估用户适配度
const answers = { /* 评估问卷答案 */ };
const userInfo = { /* 用户信息 */ };
const assessmentReport = await skillsPackage.assessUser(answers, userInfo);

// 2. 生成实施计划
const implementationPlan = skillsPackage.generateImplementationPlan('medium');

// 3. 导出报告

console.log(skillsPackage.exportPlan(implementationPlan, 'markdown'));
```
