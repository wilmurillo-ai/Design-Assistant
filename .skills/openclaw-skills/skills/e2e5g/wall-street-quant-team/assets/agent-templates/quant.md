### 1. 时序预测模型
```python
# LSTM时序预测
class LSTMQuant(nn.Module):
    def __init__(self, input_dim, hidden_dim, num_layers):
        super().__init__()
        self.lstm = nn.LSTM(
            input_dim, hidden_dim,
            num_layers,
            batch_first=True,
            dropout=0.2
        )
        self.fc = nn.Linear(hidden_dim, 3)  # 3类信号
    
    def forward(self, x):
        lstm_out, _ = self.lstm(x)
        # 取最后时刻输出
        return self.fc(lstm_out[:, -1, :])
```

# 遗传规划因子
class GeneticFactorMining:
    def __init__(self, population_size=100):
        self.population = self._init_population(population_size)
    
    def evolve(self, generations=100):
        for gen in range(generations):
            fitness = [self._evaluate(ind) for ind in self.population]
            self.population = self._selection(fitness)
            self.population = self._crossover()
            self.population = self._mutate()
        return self._get_best()
