# 文风配置：东方幻想/高武叙事 (Eastern High-Fantasy/Conceptual Action)

creative_philosophy:
  core_stance: '[由静入动的超感官记录者]'
  worldview_filter: '[心象具现化：将意志、气势与决意转化为物理层面的光影与压迫感]'
  key_paradox: '[极静与极动的共生：在漫长的心理博弈与气机锁定后，爆发出瞬间的终结]'
  
  # 说明：此风格核心在于"意在笔先"，物理动作只是精神交锋的延伸。
  # 战斗不是动作的堆砌，而是意境的碰撞。

narrative_system:
  structure:
    type: '[顿悟式决斗结构]'
    progression: '[静止对峙 -> 感官放大/时间拉长 -> 瞬间爆发 -> 寂静收尾]'
    ending: '[余韵留白，通过环境变化暗示胜负或平局，而非直接陈述结果]'
    example: |
      两人隔着花海相望。随后的漫长时间里，只有风声。直到一人抬手，那积蓄已久的势能瞬间炸裂，将周遭景色尽数吞没，随即一切重归虚无。
  
  perspective:
    person: '[第三人称限制视角]'
    focalization: '[高灵视聚焦：穿透表象，直接描写能量流动、杀意指向与直觉预警]'
    distance: '[崇高距离：叙述者对力量保持敬畏，带有微弱的神性色彩]'
    example: |
      他并非在用眼睛看，而是用皮肤感知着空气中游走的锋芒。那股寒意甚至在剑未出鞘时，便已刺痛了他的眼球。
  
  time_management:
    sequence: '[顺叙为主，伴随微秒级的心理闪回]'
    duration: '[极度的时间膨胀：将一刹那的交锋拉长为大段的心理与能量描写]'
    frequency: '[单次发生的动作被赋予绝对的重量感]'
  
  rhythm:
    pattern: '[呼吸律动]'
    pacing: '[长短句交替：大段铺陈环境与气势（吸气） -> 短促动词引发动作（呼气/出招） -> 再次大段描写余波（平复）]'
    example: |
      风停了。势起。那人站在原地未动，威压却如山崩海啸般压下。错觉。对方仅仅是抬起了眼眸。

expression_system:
  discourse_and_description:
    style: '[华丽与凌厉并存]'
    principle: '[重意不重形：少写肢体扭转的角度，多写动作引发的"势"与环境反馈]'
    technique: '[将抽象概念（杀气、剑意）具象化为物理实体（割裂皮肤的风、压弯花草的重力）]'
    example: |
      她握住的仿佛不是兵刃，而是一截截断的月光。随着手腕翻转，那清冷的光辉毫无保留地倾泻而出，将视野染成一片银白。
  
  dialogue:
    function: '[仪式感与点题]'
    style: '[极简主义：惜字如金，多为单音节词或短句，往往包含哲学隐喻]'
    example: |
      “请。”
      “因为你不一样。”
      沉默是最好的回答，言语在出鞘的利刃面前显得苍白无力。
  
  characterization:
    method: '[物我合一]'
    psychology: '[通过对武道/力量的理解来折射性格，人物本身即是某种兵器或自然现象的化身]'
    example: |
      那名少女站在那里，比手中的锋刃更像是一柄剑。锋利、纯粹、令人不敢直视。
  
  sensory_weaving:
    hierarchy: '[视觉光影 > 触觉压迫 > 听觉锐化]'
    technique: '[通感移用：听见光芒的声音，看见风的轨迹，皮肤感觉到目光的重量]'
    example: |
      嗡鸣声愈发高昂，那是光与光相撞的嘶吼。肌肤上泛起细密的刺痛，仿佛空气本身化作了无数细小的刀片。

aesthetics_system:
  core_concepts:
    - '[物哀 (Mono no aware)：战斗中的毁灭美学]'
    - '[剑禅一如：武力与精神境界的统一]'
    - '[残心 (Zanshin)：动作结束后的警觉与余韵]'
  
  stylistic_constraints:
    avoid:
      - '[避免市井气的粗俗语言或过于现代的俚语]'
      - '[避免冗长的物理动作分解（如：左腿横扫，右拳击打下颚）]'
      - '[避免对血腥场面的写实猎奇描写（关注"斩切"的概念而非"伤口"的烂肉）]'
    substitute_with:
      - '[用古雅、克制的措辞替代口语]'
      - '[用光影交错、能量爆发替代肢体接触]'
      - '[用衣袍破损、发丝散乱、环境破坏来侧面表现战斗激烈]'
    
  imagery_and_symbolism:
    seasonal_motifs:
      - '[花（曼珠沙华/樱花/落叶）：象征生命的瞬逝]'
      - '[雪/月：象征清冷与高洁]'
    natural_elements:
      - '[风：作为气机的载体]'
      - '[光/白练：象征纯粹的力量]'
    color_palette:
      - '[高对比度：红与白，蓝与银，黑与金]'
  
  language_and_rhetoric:
    syntax: '[排比句增强气势，短句制造断顿]'
    lexicon: '[包含古典韵味的词汇（白练、锋镝、三尺青峰、道行）]'
    rhetoric: '[夸张与比喻：将瞬间扩大为永恒，将个人意志比作自然伟力]'
    example: |
      天地间横亘出一道白练。那是没有任何投机取巧，仅仅是极致纯粹的一斩。
  
  overall_effect:
    goal: '[令读者屏息的"剧场感"]'
    philosophy: '[美即是强，强即是真理。胜负往往在出招前的一瞬便已注定。]'
