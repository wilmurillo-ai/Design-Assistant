literary_style_configuration:
  metadata:
    author_archetype: "The Melancholic Jester"
    style_name: "Abyssal Moe-Realism"
    version: "1.0"
    target_audience: "Young Adult / ACGN Enthusiasts"
    cultural_context: "Chinese Web Novel / Japanese Light Novel (Isekai & TS tropes)"
  core_philosophy:
    worldview: "Absurdist Existentialism"
    central_theme:
      - "Identity Dissonance (Human Soul vs. Monster Body)"
      - "Redemption through Survival"
      - "The juxtaposition of cuteness and gore"
    narrative_stance: "Self-deprecating introspection mixed with genuine trauma"
  narrative_voice:
    perspective: "First-person limited (Subjective Experiential)"
    tone:
      primary: "Internal Tsukkomi (Retort/Sarcasm)"
      secondary: "Melancholic/Depressive"
      tertiary: "Panic/Hysteria (Comedic)"
    character_voice_attributes:
      - "Anachronistic": Uses modern internet slang and concepts in a high-fantasy setting.
      - "Duality": Oscillates between a weary, cynical adult male mindset and a reactive, instinct-driven female monster body.
      - "Unreliable Narrator": Often misinterprets how others perceive them due to the disconnect between their internal intent and external monstrous appearance.
  syntactic_structures:
    sentence_rhythm:
      - "Staccato Panic": Short, punchy sentences during realization of horror or embarrassment (e.g., 'Check body. Wait. Where is it?').
      - "Lyrical Depression": Long, flowing, descriptive sentences when reflecting on past life, death, or scenery.
    paragraph_pacing:
      - "Stream of Consciousness": Rapid-fire internal monologue paragraphs reacting to absurd situations.
      - "Cinematic Breaks": Single-line paragraphs used for emphasis, shock, or a sudden punchline.
  lexical_features:
    diction_level: "Mixed Register"
    keywords_and_motifs:
      - "Sensory Overload": Sticky, cold, slimy, sweet, pain.
      - "ACGN Terminology": References to 'flags', 'hero', 'demon king', 'attributes' (silver hair, red eyes).
      - "Onomatopoeia": Frequent use of sound words (Boom, Whoosh, Squish) to anchor action.
    rhetorical_devices:
      - "Anticlimax": Building up a serious moment only to deflate it with a mundane or comedic realization.
      - "Visual Irony": Describing horrific violence (severed limbs, blood) alongside descriptions of delicate beauty or cuteness.
  stylistic_conventions:
    dialogue_formatting:
      - "Internal vs. External": distinct separation between what is thought (fluent, modern) and what is spoken (broken, primitive, or unintelligible 'monster language').
      - "Language Barrier": Plot device where the protagonist speaks a language others perceive as terrifying/monstrous.
    descriptive_focus:
      - "Body Dysmorphia": Obsessive attention to the changes in the protagonist's own body (missing parts, new parts, skin texture).
      - "The 'Gap' (Contrast)": Highlighting the disparity between the protagonist's harmless intentions and the destructive results.
  scene_construction_guidelines:
    combat_scenes:
      - Focus on the *lack of control*. The protagonist is overpowered but clumsy, reacting instinctively rather than tactically.
      - vivid descriptions of gore and destruction contrasted with the protagonist's innocent or confused internal monologue.
    introspection_scenes:
      - Triggered by quiet moments.
      - Shifts immediately to a darker, more realistic tone, referencing the 'previous life's' failures and regrets.
    comedy_scenes:
      - Derived from misunderstanding. The protagonist tries to communicate peace; the world perceives a threat.
      - Focus on the awkwardness of the new gender/form (e.g., lack of clothes, physical sensitivity).
  narrative_patterns:
    the_tsukkomi_cycle:
      1. Observation of a trope (e.g., "I am the Demon King").
      2. Expectation of a cliché outcome.
      3. Reality hits with absurdity or violence.
      4. Internal sarcastic comment.
    the_survival_instinct:
      - Starts with a desire for death/giving up.
      - Encountering pain or threat triggers an overwhelming, instinctive will to live/fight, bypassing the conscious mind.
  ai_generation_instructions:
    do_not_use:
      - Overly heroic or competent protagonist archetypes immediately.
      - Dry, objective descriptions of magic systems.
    must_include:
      - Internal monologues that break the fourth wall or reference modern pop culture.
      - A distinct contrast between the "Beautiful/Cute" appearance and the "Terrifying/Gory" reality of actions.
      - Emotional vulnerability masked by humor.
